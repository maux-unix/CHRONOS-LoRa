21 class MagickDLLDecl Montage
22 {
23 public:
24 Montage( void );
25 virtual ~Montage( void );
26
27 // Specifies the background color that thumbnails are imaged upon.
28 void backgroundColor ( const Color &backgroundColor_ );
29 Color backgroundColor ( void ) const;
30
31 // Specifies the image composition algorithm for thumbnails. This
32 // controls the algorithm by which the thumbnail image is placed
33 // on the background. Use of OverCompositeOp is recommended for
34 // use with images that have transparency. This option may have
35 // negative side-effects for images without transparency.
36 void compose ( CompositeOperator compose_ );
37 CompositeOperator compose ( void ) const;
38
39 // Specifies the image filename to be used for the generated
40 // montage images. To handle the case were multiple montage images
41 // are generated, a printf-style format may be embedded within the
42 // filename. For example, a filename specification of
43 // image%02d.miff names the montage images as image00.miff,
44 // image01.miff, etc.
45 void fileName( const std::string &fileName_ );
46 std::string fileName( void ) const;
47
48 // Specifies the fill color to use for the label text.
49 void fillColor ( const Color &fill_ );
50 Color fillColor ( void ) const;
51
52 // Specifies the thumbnail label font.
53 void font ( const std::string &font_ );
54 std::string font ( void ) const;
55
56 // Specifies the size of the generated thumbnail.
57 void geometry ( const Geometry &geometry_ );
58 Geometry geometry ( void ) const;
59
60 // Specifies the thumbnail positioning within the specified
61 // geometry area. If the thumbnail is smaller in any dimension
62 // than the geometry, then it is placed according to this
63 // specification
64 void gravity ( GravityType gravity_ );
65 GravityType gravity ( void ) const;
66
67 // Specifies the format used for the image label. Special format
68 // characters may be embedded in the format string to include
69 // information about the image.
70 void label( const std::string &label_ );
71 std::string label( void ) const;
72
73 // Specifies the pen color to use for the label text (same as fill).
74 void penColor ( const Color &pen_ );
75 Color penColor ( void ) const;
76
77 // Specifies the thumbnail label font size.
78 void pointSize ( unsigned int pointSize_ );
79 unsigned int pointSize ( void ) const;
80
81 // Enable/disable drop-shadow on thumbnails.
82 void shadow ( bool shadow_ );
83 bool shadow ( void ) const;
84
85 // Specifies the stroke color to use for the label text .
86 void strokeColor ( const Color &stroke_ );
87 Color strokeColor ( void ) const;
88
89 // Specifies a texture image to use as montage background. The
90 // built-in textures "granite:" and "plasma:" are available. A
91 // texture is the same as a background image.
92 void texture ( const std::string &texture_ );
93 std::string texture ( void ) const;
94
95 // Specifies the maximum number of montage columns and rows in the
96 // montage. The montage is built by filling out all cells in a row
97 // before advancing to the next row. Once the montage has reached
98 // the maximum number of columns and rows, a new montage image is
99 // started.
100 void tile ( const Geometry &tile_ );
101 Geometry tile ( void ) const;
102
103 // Specifies the montage title
104 void title ( const std::string &title_ );
105 std::string title ( void ) const;
106
107 // Specifies a montage color to set transparent. This option can
108 // be set the same as the background color in order for the
109 // thumbnails to appear without a background when rendered on an
110 // HTML page. For best effect, ensure that the transparent color
111 // selected does not occur in the rendered thumbnail colors.
112 void transparentColor ( const Color &transparentColor_ );
113 Color transparentColor ( void ) const;
114
115 //
116 // Implementation methods/members
117 //
118
119 // Update elements in existing MontageInfo structure
120 virtual void updateMontageInfo ( MagickLib::MontageInfo &montageInfo_ ) const;
121
122 protected:
123
124 private:
125 Color _backgroundColor; // Color that thumbnails are composed on
126 CompositeOperator _compose; // Composition algorithm to use (e.g. ReplaceCompositeOp)
127 std::string _fileName; // Filename to save montages to
128 Color _fill; // Fill color
129 std::string _font; // Label font
130 Geometry _geometry; // Thumbnail width & height plus border width & height
131 GravityType _gravity; // Thumbnail position (e.g. SouthWestGravity)
132 std::string _label; // Thumbnail label (applied to image prior to montage)
133 unsigned int _pointSize; // Font point size
134 bool _shadow; // Enable drop-shadows on thumbnails
135 Color _stroke; // Outline color
136 std::string _texture; // Background texture image
137 Geometry _tile; // Thumbnail rows and colmns
138 std::string _title; // Montage title
139 Color _transparentColor; // Transparent color
140 };

154
155 // Specifies the border (in pixels) to place between a thumbnail
156 // and its surrounding frame. This option only takes effect if
157 // thumbnail frames are enabled (via frameGeometry) and the

162

166 void frameGeometry ( const Geometry &frame_ );
167 Geometry frameGeometry ( void ) const;
168
169 // Specifies the thumbnail frame color.
170 void matteColor ( const Color &matteColor_ );
171 Color matteColor ( void ) const;
172
173 //
174 // Implementation methods/members
175 //
176
177 // Update elements in existing MontageInfo structure
178 /* virtual */ void updateMontageInfo ( MagickLib::MontageInfo &montageInfo_ ) const;
179
180 protected:
181
182 private:
183
184 Color _borderColor; // Frame border color
185 unsigned int _borderWidth; // Pixels between thumbnail and surrounding frame
186 Geometry _frame; // Frame geometry (width & height frame thickness)
187 Color _matteColor; // Frame foreground color
188 };
