668inline OptionNames split_option_names(const std::string &text)
669{
670 OptionNames split_names;
671
672 std::string::size_type token_start_pos = 0;
673 auto length = text.length();
674
675 if (length == 0)
676 {
677 throw_or_mimic<exceptions::invalid_option_format>(text);
678 }
679
680 while (token_start_pos < length) {
681 const auto &npos = std::string::npos;
682 auto next_non_space_pos = text.find_first_not_of(' ', token_start_pos);
683 if (next_non_space_pos == npos) {
684 throw_or_mimic<exceptions::invalid_option_format>(text);
685 }
686 token_start_pos = next_non_space_pos;
687 auto next_delimiter_pos = text.find(',', token_start_pos);
688 if (next_delimiter_pos == token_start_pos) {
689 throw_or_mimic<exceptions::invalid_option_format>(text);
690 }
691 if (next_delimiter_pos == npos) {
692 next_delimiter_pos = length;
693 }
694 auto token_length = next_delimiter_pos - token_start_pos;
695 // validate the token itself matches the regex /([:alnum:][-_[:alnum:]]*/
696 {
697 const char* option_name_valid_chars =
698 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
699 "abcdefghijklmnopqrstuvwxyz"
700 "0123456789"
701 "_-.?";
702
703 if (!std::isalnum(text[token_start_pos], std::locale::classic()) ||
704 text.find_first_not_of(option_name_valid_chars, token_start_pos) < next_delimiter_pos) {
705 throw_or_mimic<exceptions::invalid_option_format>(text);
706 }
707 }
708 split_names.emplace_back(text.substr(token_start_pos, token_length));
709 token_start_pos = next_delimiter_pos + 1;
710 }
711 return split_names;
712}

714inline ArguDesc ParseArgument(const char *arg, bool &matched)
715{
716 ArguDesc argu_desc;
717 const char *pdata = arg;
718 matched = false;
719 if (strncmp(pdata, "--", 2) == 0)
720 {
721 pdata += 2;
722 if (isalnum(*pdata, std::locale::classic()))
723 {
724 argu_desc.arg_name.push_back(*pdata);
725 pdata += 1;
726 while (isalnum(*pdata, std::locale::classic()) || *pdata == '-' || *pdata == '_')
727 {
728 argu_desc.arg_name.push_back(*pdata);
729 pdata += 1;
730 }
731 if (argu_desc.arg_name.length() > 1)
732 {
733 if (*pdata == '=')
734 {
735 argu_desc.set_value = true;
736 pdata += 1;
737 if (*pdata != '\0')
738 {
739 argu_desc.value = std::string(pdata);
740 }
741 matched = true;
742 }
743 else if (*pdata == '\0')
744 {
745 matched = true;
746 }
747 }
748 }
749 }
750 else if (strncmp(pdata, "-", 1) == 0)
751 {
752 pdata += 1;
753 argu_desc.grouping = true;
754 while (isalnum(*pdata, std::locale::classic()))
755 {
756 argu_desc.arg_name.push_back(*pdata);
757 pdata += 1;
758 }
759 matched = !argu_desc.arg_name.empty() && *pdata == '\0';
760 }
761 return argu_desc;
762}

787inline IntegerDesc SplitInteger(const std::string &text)
788{
789 static const std::basic_regex<char> integer_matcher(integer_pattern);
790
791 std::smatch match;
792 std::regex_match(text, match, integer_matcher);
793
794 if (match.length() == 0)
795 {
796 throw_or_mimic<exceptions::incorrect_argument_type>(text);
797 }
798
799 IntegerDesc desc;
800 desc.negative = match[1];
801 desc.base = match[2];
802 desc.value = match[3];
803
804 if (match.length(4) > 0)
805 {
806 desc.base = match[5];
807 desc.value = "0";
808 return desc;
809 }
810
811 return desc;
812}

830// Gets the option names specified via a single, comma-separated string,
831// and returns the separate, space-discarded, non-empty names
832// (without considering which or how many are single-character)
833inline OptionNames split_option_names(const std::string &text)
834{
835 static const std::basic_regex<char> option_specifier_matcher(option_specifier_pattern);
836 if (!std::regex_match(text.c_str(), option_specifier_matcher))
837 {
838 throw_or_mimic<exceptions::invalid_option_format>(text);
839 }
840
841 OptionNames split_names;
842
843 static const std::basic_regex<char> option_specifier_separator_matcher(option_specifier_separator_pattern);
844 constexpr int use_non_matches { -1 };
845 auto token_iterator = std::sregex_token_iterator(
846 text.begin(), text.end(), option_specifier_separator_matcher, use_non_matches);
847 std::copy(token_iterator, std::sregex_token_iterator(), std::back_inserter(split_names));
848 return split_names;
849}

851inline ArguDesc ParseArgument(const char *arg, bool &matched)
852{
853 static const std::basic_regex<char> option_matcher(option_pattern);
854 std::match_results<const char*> result;
855 std::regex_match(arg, result, option_matcher);
856 matched = !result.empty();
857
858 ArguDesc argu_desc;
859 if (matched) {
860 argu_desc.arg_name = result[1].str();
861 argu_desc.set_value = result[2].length() > 0;
862 argu_desc.value = result[3].str();
863 if (result[4].length() > 0)
864 {
865 argu_desc.grouping = true;
866 argu_desc.arg_name = result[4].str();

940template <typename T>
941void
942integer_parser(const std::string& text, T& value)
943{
944 parser_tool::IntegerDesc int_desc = parser_tool::SplitInteger(text);
945
946 using US = typename std::make_unsigned<T>::type;
947 constexpr bool is_signed = std::numeric_limits<T>::is_signed;
948
949 const bool negative = int_desc.negative.length() > 0;
950 const uint8_t base = int_desc.base.length() > 0 ? 16 : 10;
951 const std::string & value_match = int_desc.value;
952
953 US result = 0;
954
955 for (char ch : value_match)
956 {
957 US digit = 0;
958
959 if (ch >= '0' && ch <= '9')
960 {
961 digit = static_cast<US>(ch - '0');
962 }
963 else if (base == 16 && ch >= 'a' && ch <= 'f')
964 {
965 digit = static_cast<US>(ch - 'a' + 10);
966 }
967 else if (base == 16 && ch >= 'A' && ch <= 'F')
968 {
969 digit = static_cast<US>(ch - 'A' + 10);
970 }
971 else
972 {
973 throw_or_mimic<exceptions::incorrect_argument_type>(text);
974 }
975
976 US limit = 0;
977 if (negative)
978 {
979 limit = static_cast<US>(std::abs(static_cast<intmax_t>(std::numeric_limits<T>::min())));
980 }
981 else
982 {
983 limit = std::numeric_limits<T>::max();
984 }
985
986 if (base != 0 && result > limit / base)
987 {
988 throw_or_mimic<exceptions::incorrect_argument_type>(text);
989 }
990 if (result * base > limit - digit)
991 {
992 throw_or_mimic<exceptions::incorrect_argument_type>(text);
993 }
994
995 result = static_cast<US>(result * base + digit);
996 }
997
998 detail::check_signed_range<T>(negative, result, text);
999
1000 if (negative)
1001 {
1002 checked_negate<T>(value, result, text, std::integral_constant<bool, is_signed>());
1003 }
1004 else
1005 {
1006 value = static_cast<T>(result);
1007 }
1008}

2724{

2726
2727 auto group = m_help.find(g);
2728 if (group == m_help.end())
2729 {
2730 return "";
2731 }
2732
2733 OptionHelp format;
2734
2735 std::size_t longest = 0;
2736
2737 String result;
2738
2739 if (!g.empty())
2740 {
2741 result += toLocalString(" " + g + " options:\n");
2742 }
2743
2744 for (const auto& o : group->second.options)

2747 m_positional_set.find(o.l.front()) != m_positional_set.end() &&
2748 !m_show_positional)
2749 {
2750 continue;
2751 }
2752
2753 auto s = format_option(o);
2754 longest = (std::max)(longest, stringLength(s));
2755 format.push_back(std::make_pair(s, String()));
2756 }
2757 longest = (std::min)(longest, OPTION_LONGEST);
2758
2759 //widest allowed description -- min 10 chars for helptext/line
2760 std::size_t allowed = 10;
2761 if (m_width > allowed + longest + OPTION_DESC_GAP)
2762 {
2763 allowed = m_width - longest - OPTION_DESC_GAP;
2764 }
2765
2766 auto fiter = format.begin();
2767 for (const auto& o : group->second.options)
2768 {
2769 if (o.l.size() &&
2770 m_positional_set.find(o.l.front()) != m_positional_set.end() &&
2771 !m_show_positional)
2772 {
2773 continue;
2774 }
2775
2776 auto d = format_description(o, longest + OPTION_DESC_GAP, allowed, m_tab_expansion);
2777
2778 result += fiter->first;
2779 if (stringLength(fiter->first) > longest)
2780 {
2781 result += '\n';
2782 result += toLocalString(std::string(longest + OPTION_DESC_GAP, ' '));
2783 }
2784 else
2785 {
2786 result += toLocalString(std::string(longest + OPTION_DESC_GAP -
2787 stringLength(fiter->first),
2788 ' '));
2789 }
2790 result += d;
2791 result += '\n';

2793 ++fiter;
