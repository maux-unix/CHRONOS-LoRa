35/**
36 * @brief Compute the CRC32C checksum of a memory span.
37 *
38 * Uses ARM hardware-accelerated CRC32C byte intrinsics for efficient
39 * byte-wise checksum computation.
40 *
41 * Initial CRC seed is `0xFFFFFFFF`, and the final result is XOR'd with
42 * `0xFFFFFFFF` per CRC32C standard conventions.
43 *
44 * Debug information is printed before and after processing.
45 *
46 * @param frame The memory frame to process.
47 *
48 * @return The finalized CRC32C checksum.
49 *
50 * @note Requires ARMv8 or newer architecture with CRC32 instruction support.
51 */
52[[nodiscard]] inline std::uint32_t
53encode(const Frame& frame, std::uint32_t previous_crc = 0)
54{
55 std::uint32_t crc = ~previous_crc;
56 const auto *bytes = frame.data;
57
58 for (std::size_t i = 0; i < frame.length; ++i) {
59 crc = __crc32cb(crc, bytes[i]);
60 }
61
62 return ~crc;
63}
