102template <typename Char, typename PathChar>
103auto get_path_string(const std::filesystem::path& p,
104 const std::basic_string<PathChar>& native) {
105 if constexpr (std::is_same_v<Char, char> &&
106 std::is_same_v<PathChar, wchar_t>) {
107 return to_utf8<wchar_t>(native, to_utf8_error_policy::wtf);
108 } else {
109 return p.string<Char>();
110 }
111}
112
113template <typename Char, typename PathChar>
114void write_escaped_path(basic_memory_buffer<Char>& quoted,
115 const std::filesystem::path& p,
116 const std::basic_string<PathChar>& native) {
117 if constexpr (std::is_same_v<Char, char> &&
118 std::is_same_v<PathChar, wchar_t>) {
119 auto buf = basic_memory_buffer<wchar_t>();
120 write_escaped_string<wchar_t>(std::back_inserter(buf), native);
121 bool valid = to_utf8<wchar_t>::convert(quoted, {buf.data(), buf.size()});
122 FMT_ASSERT(valid, "invalid utf16");
123 } else if constexpr (std::is_same_v<Char, PathChar>) {
124 write_escaped_string<std::filesystem::path::value_type>(
125 std::back_inserter(quoted), native);
126 } else {
127 write_escaped_string<Char>(std::back_inserter(quoted), p.string<Char>());
128 }
129}

166#if FMT_USE_RTTI
167inline auto normalize_libcxx_inline_namespaces(string_view demangled_name_view,
168 char* begin) -> string_view {
169 // Normalization of stdlib inline namespace names.
170 // libc++ inline namespaces.
171 // std::__1::* -> std::*
172 // std::__1::__fs::* -> std::*
173 // libstdc++ inline namespaces.
174 // std::__cxx11::* -> std::*
175 // std::filesystem::__cxx11::* -> std::filesystem::*
176 if (demangled_name_view.starts_with("std::")) {
177 char* to = begin + 5; // std::
178 for (const char *from = to, *end = begin + demangled_name_view.size();
179 from < end;) {
180 // This is safe, because demangled_name is NUL-terminated.
181 if (from[0] == '_' && from[1] == '_') {
182 const char* next = from + 1;
183 while (next < end && *next != ':') next++;
184 if (next[0] == ':' && next[1] == ':') {
185 from = next + 2;
186 continue;
187 }
188 }
189 *to++ = *from++;
190 }
191 demangled_name_view = {begin, detail::to_unsigned(to - begin)};
192 }
193 return demangled_name_view;
194}
195
196template <class OutputIt>
197auto normalize_msvc_abi_name(string_view abi_name_view, OutputIt out)
198 -> OutputIt {
199 const string_view demangled_name(abi_name_view);
200 for (size_t i = 0; i < demangled_name.size(); ++i) {
201 auto sub = demangled_name;
202 sub.remove_prefix(i);
203 if (sub.starts_with("enum ")) {
204 i += 4;
205 continue;
206 }
207 if (sub.starts_with("class ") || sub.starts_with("union ")) {
208 i += 5;
209 continue;
210 }
211 if (sub.starts_with("struct ")) {
212 i += 6;
213 continue;
214 }
215 if (*sub.begin() != ' ') *out++ = *sub.begin();
216 }
217 return out;
218}

220template <typename OutputIt>
221auto write_demangled_name(OutputIt out, const std::type_info& ti) -> OutputIt {
222# ifdef FMT_HAS_ABI_CXA_DEMANGLE
223 int status = 0;
224 size_t size = 0;
225 std::unique_ptr<char, void (*)(void*)> demangled_name_ptr(
226 abi::__cxa_demangle(ti.name(), nullptr, &size, &status), &free);
227
228 string_view demangled_name_view;
229 if (demangled_name_ptr) {
230 demangled_name_view = normalize_libcxx_inline_namespaces(
231 demangled_name_ptr.get(), demangled_name_ptr.get());
232 } else {
233 demangled_name_view = string_view(ti.name());
234 }
235 return detail::write_bytes<char>(out, demangled_name_view);
236# elif FMT_MSC_VERSION && defined(_MSVC_STL_UPDATE)
237 return normalize_msvc_abi_name(ti.name(), out);
238# elif FMT_MSC_VERSION && defined(_LIBCPP_VERSION)
239 const string_view demangled_name = ti.name();
240 std::string name_copy(demangled_name.size(), '\0');
241 // normalize_msvc_abi_name removes class, struct, union etc that MSVC has in
242 // front of types
243 name_copy.erase(normalize_msvc_abi_name(demangled_name, name_copy.begin()),
244 name_copy.end());
245 // normalize_libcxx_inline_namespaces removes the inline __1, __2, etc
246 // namespaces libc++ uses for ABI versioning On MSVC ABI + libc++
247 // environments, we need to eliminate both of them.
248 const string_view normalized_name =
249 normalize_libcxx_inline_namespaces(name_copy, name_copy.data());
250 return detail::write_bytes<char>(out, normalized_name);
251# else
252 return detail::write_bytes<char>(out, string_view(ti.name()));
253# endif
254}

305template <typename Char> struct formatter<std::filesystem::path, Char> {
306 private:
307 format_specs specs_;
308 detail::arg_ref<Char> width_ref_;
309 bool debug_ = false;
310 char path_type_ = 0;
311
312 public:
313 FMT_CONSTEXPR void set_debug_format(bool set = true) { debug_ = set; }
314
315 FMT_CONSTEXPR auto parse(parse_context<Char>& ctx) {
316 auto it = ctx.begin(), end = ctx.end();
317 if (it == end) return it;
318
319 it = detail::parse_align(it, end, specs_);
320 if (it == end) return it;
321
322 Char c = *it;
323 if ((c >= '0' && c <= '9') || c == '{')
324 it = detail::parse_width(it, end, specs_, width_ref_, ctx);
325 if (it != end && *it == '?') {
326 debug_ = true;
327 ++it;
328 }
329 if (it != end && (*it == 'g')) path_type_ = detail::to_ascii(*it++);
330 return it;
331 }
332
333 template <typename FormatContext>
334 auto format(const std::filesystem::path& p, FormatContext& ctx) const {
335 auto specs = specs_;
336 auto path_string =
337 !path_type_ ? p.native()
338 : p.generic_string<std::filesystem::path::value_type>();
339
340 detail::handle_dynamic_spec(specs.dynamic_width(), specs.width, width_ref_,
341 ctx);
342 if (!debug_) {
343 auto s = detail::get_path_string<Char>(p, path_string);
344 return detail::write(ctx.out(), basic_string_view<Char>(s), specs);
345 }
346 auto quoted = basic_memory_buffer<Char>();
347 detail::write_escaped_path(quoted, p, path_string);
348 return detail::write(ctx.out(),
349 basic_string_view<Char>(quoted.data(), quoted.size()),
350 specs);
351 }
352};

550 private:
551 format_specs specs_;
552 detail::arg_ref<char> width_ref_;

556 FMT_CONSTEXPR void set_debug_format(bool set = true) { debug_ = set; }

558 FMT_CONSTEXPR auto parse(parse_context<>& ctx) -> const char* {
559 auto it = ctx.begin(), end = ctx.end();
560 if (it == end) return it;

566 it = detail::parse_width(it, end, specs_, width_ref_, ctx);
567
568 if (it != end && *it == '?') {
569 debug_ = true;

579 template <typename FormatContext>
580 FMT_CONSTEXPR20 auto format(const std::error_code& ec,
581 FormatContext& ctx) const -> decltype(ctx.out()) {
582 auto specs = specs_;
583 detail::handle_dynamic_spec(specs.dynamic_width(), specs.width, width_ref_,
584 ctx);
585 auto buf = memory_buffer();
586 if (specs_.type() == presentation_type::string) {
587 buf.append(ec.message());
588 } else {
589 buf.append(string_view(ec.category().name()));
590 buf.push_back(':');
591 detail::write<char>(appender(buf), ec.value());
592 }
593 auto quoted = memory_buffer();
594 auto str = string_view(buf.data(), buf.size());
595 if (debug_) {
596 detail::write_escaped_string<char>(std::back_inserter(quoted), str);
597 str = string_view(quoted.data(), quoted.size());
598 }
599 return detail::write<char>(ctx.out(), str, specs);
600 }

730template <typename T, typename Char> struct formatter<std::complex<T>, Char> {
731 private:
732 detail::dynamic_format_specs<Char> specs_;
733
734 template <typename FormatContext, typename OutputIt>
735 FMT_CONSTEXPR auto do_format(const std::complex<T>& c,
736 detail::dynamic_format_specs<Char>& specs,
737 FormatContext& ctx, OutputIt out) const
738 -> OutputIt {
739 if (c.real() != 0) {
740 *out++ = Char('(');
741 out = detail::write<Char>(out, c.real(), specs, ctx.locale());
742 specs.set_sign(sign::plus);
743 out = detail::write<Char>(out, c.imag(), specs, ctx.locale());
744 if (!detail::isfinite(c.imag())) *out++ = Char(' ');
745 *out++ = Char('i');
746 *out++ = Char(')');
747 return out;
748 }
749 out = detail::write<Char>(out, c.imag(), specs, ctx.locale());
750 if (!detail::isfinite(c.imag())) *out++ = Char(' ');
751 *out++ = Char('i');
752 return out;
753 }
754
755 public:
756 FMT_CONSTEXPR auto parse(parse_context<Char>& ctx) -> const Char* {
757 if (ctx.begin() == ctx.end() || *ctx.begin() == '}') return ctx.begin();
758 return parse_format_specs(ctx.begin(), ctx.end(), specs_, ctx,
759 detail::type_constant<T, Char>::value);
760 }
761
762 template <typename FormatContext>
763 auto format(const std::complex<T>& c, FormatContext& ctx) const
764 -> decltype(ctx.out()) {
765 auto specs = specs_;
766 if (specs.dynamic()) {
767 detail::handle_dynamic_spec(specs.dynamic_width(), specs.width,
768 specs.width_ref, ctx);
769 detail::handle_dynamic_spec(specs.dynamic_precision(), specs.precision,
770 specs.precision_ref, ctx);
771 }
772
773 if (specs.width == 0) return do_format(c, specs, ctx, ctx.out());
774 auto buf = basic_memory_buffer<Char>();
775
776 auto outer_specs = format_specs();
777 outer_specs.width = specs.width;
778 outer_specs.copy_fill_from(specs);
779 outer_specs.set_align(specs.align());
780
781 specs.width = 0;
782 specs.set_fill({});
783 specs.set_align(align::none);
784
785 do_format(c, specs, ctx, basic_appender<Char>(buf));
786 return detail::write<Char>(ctx.out(),
787 basic_string_view<Char>(buf.data(), buf.size()),
788 outer_specs);
789 }
790};
