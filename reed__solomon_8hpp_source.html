45inline int
46rs_init(rs_ctx_t *ctx)
47{
48 ctx->rs = correct_reed_solomon_create(

59inline void
60rs_encode(rs_ctx_t *ctx, const uint8_t *in, uint8_t *out)

62 memcpy(out, in, RS_K);

72inline void
73rs_example(void)
74{
75 correct_reed_solomon *rs = correct_reed_solomon_create(
76 correct_rs_primitive_polynomial_8_4_3_2_0, // primitive_polynomial
77 1, // first_consecutive_root
78 1, // generator_root_gap
79 NROOTS // num_roots
80 );
81
82 if (!rs) {
83 printf("Failed to create RS codec");
84 }
85
86 uint8_t data[RS_K];
87 uint8_t encoded[RS_N];
88 uint8_t decoded[RS_K];
89
90 memset(data, 0, RS_K);
91 strcpy((char *)data, "HELLO LORA");
92
93 memcpy(encoded, data, RS_K);
94
95 correct_reed_solomon_encode(rs, data, RS_K, encoded + RS_K);
96
97 printf("RS encoded\n");
98
99 // Simulate corruption
100 encoded[5] ^= 0xFF;
101 encoded[100] ^= 0xFF;
102
103 auto rs_result = correct_reed_solomon_decode(rs, encoded, RS_N, decoded);
104
105 if (rs_result < 0) {
106 printf("Decode failed");
107 } else {
108 printf("Decoded %ld bytes\n", rs_result);
109 printf("Message: %.*s\n", (int)rs_result, decoded);
110 }
111
112 // printf("Corrected errors: %d\n", corrected);
113
114 correct_reed_solomon_destroy(rs);
