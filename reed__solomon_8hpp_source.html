37int
38rs_init(rs_ctx_t *ctx)
39{
40 ctx->rs = correct_reed_solomon_create(

42 return ctx->rs ? 0 : -1;

64void
65rs_example(void)
66{
67 correct_reed_solomon *rs = correct_reed_solomon_create(
68 correct_rs_primitive_polynomial_8_4_3_2_0, // primitive_polynomial
69 1, // first_consecutive_root
70 1, // generator_root_gap
71 NROOTS // num_roots
72 );
73
74 if (!rs) {
75 printf("Failed to create RS codec");
76 }
77
78 uint8_t data[RS_K];
79 uint8_t encoded[RS_N];
80 uint8_t decoded[RS_K];
81
82 memset(data, 0, RS_K);
83 memcpy(data, "HELLO LORA", 10);
84
85 memcpy(encoded, data, RS_K);
86
87 correct_reed_solomon_encode(rs, data, RS_K, encoded + RS_K);
88
89 printf("RS encoded\n");
90
91 // Simulate corruption
92 encoded[5] ^= 0xFF;
93 encoded[100] ^= 0xFF;
94
95 auto rs_result = correct_reed_solomon_decode(rs, encoded, RS_N, decoded);
96
97 if (rs_result < 0) {
98 printf("Decode failed");
99 } else {
100 printf("Decoded %ld bytes\n", rs_result);
101 printf("Message: %.*s\n", (int)rs_result, decoded);
102 }
103
104 // printf("Corrected errors: %d\n", corrected);
105
106 correct_reed_solomon_destroy(rs);
107}
