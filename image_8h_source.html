95/*
96 MaxMap defines the maximum index value for algorithms which depend
97 on lookup tables (e.g. colorspace transformations and
98 normalization). When MaxMap is less than MaxRGB it is necessary to
99 downscale samples to fit the range of MaxMap. The number of bits
100 which are effectively preserved depends on the size of MaxMap.
101 MaxMap should be a multiple of 255 and no larger than MaxRGB. Note
102 that tables can become quite large and as the tables grow larger it
103 may take more time to compute the table than to process the image.
104*/
105#define MaxMap 65535U
106#define MaxMapDepth 16
107#define MaxMapFloat 65535.0f
108#define MaxMapDouble 65535.0
